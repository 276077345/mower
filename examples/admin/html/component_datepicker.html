$(function() {
    $('#datetimepicker1').datetimepicker();

    $('#datetimepicker2').datetimepicker({
        format: 'LT'
    });

    $('#datetimepicker3').datetimepicker({
        daysOfWeekDisabled: [0, 6]
    });

    $('#datetimepicker4').datetimepicker({
        inline: true,
        sideBySide: true
    });
});
